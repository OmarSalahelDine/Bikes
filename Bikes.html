from geopy.distance import great_circle
def dist_kilos(y):
    loc_1 = (y['start_station_latitude'],y['start_station_longitude']) 
    loc_2 = (y['end_station_latitude'],y['end_station_longitude'])
    return great_circle(loc_1,loc_2).kilometers

df['dist_in_kilos'] = df.apply(dist_kilos , axis = 1)
